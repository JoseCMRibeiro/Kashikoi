export class Slideshow { constructor() { this.slides = document.getElementsByClassName("mySlides"); this.timeout = null; this.isRunning = false; } start() { if (!this.isRunning) { this.isRunning = true; this.carousel(); } } stop() { clearTimeout(this.timeout); this.isRunning = false; } restart() { this.stop(); this.start(); } carousel() { this.timeout = setTimeout(() => { this.carousel(); }, 2000); for (let i = 0; i < this.slides.length; i++) { if (this.slides[0].style.display === "") { this.slides[0].style.display = "block"; } else if (this.slides[i].style.display === "block") { this.slides[i].style.display = "none"; if (i < this.slides.length - 1) { this.slides[i + 1].style.display = "block"; i = this.slides.length; } else { this.slides[0].style.display = "block"; } } } } isSlideshowRunning() { return this.isRunning; } }
